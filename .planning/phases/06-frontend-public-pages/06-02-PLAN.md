    3. Update frontend/src/app/page.tsx to assemble all sections with API data:
       ```typescript
       import { fetchAPI } from '@/lib/api'
       import { SITE_NAME, SITE_URL } from '@/lib/constants'
       import { HeroSection } from '@/components/sections/Hero'
       import { CoursesShowcaseSection } from '@/components/sections/CoursesShowcase'
       import { CommunityFeaturesSection } from '@/components/sections/CommunityFeatures'
       import { TestimonialsSection } from '@/components/sections/Testimonials'
       import { FAQSection } from '@/components/sections/FAQ'
       import type { Course } from '@/types'

       interface CoursesResponse {
         courses: Course[]
       }

       async function getCourses(): Promise<Course[]> {
         try {
           const data = await fetchAPI<CoursesResponse>('/courses', {
             revalidate: 3600, // Revalidate every hour
           })
           return data.courses
         } catch (error) {
           console.error('Failed to fetch courses:', error)
           return []
         }
       }

       export default async function HomePage() {
         const courses = await getCourses()

         const jsonLd = {
           '@context': 'https://schema.org',
           '@type': 'Organization',
           name: SITE_NAME,
           description: 'Investment education platform for Indonesian investors',
           url: SITE_URL,
           logo: `${SITE_URL}/logo.png`,
           sameAs: ['https://discord.gg/stockus'],
         }

         return (
           <>
             <script
               type="application/ld+json"
               dangerouslySetInnerHTML={{ __html: JSON.stringify(jsonLd) }}
             />
             <HeroSection />
             <CoursesShowcaseSection courses={courses} />
             <CommunityFeaturesSection />
             <TestimonialsSection />
             <FAQSection />
           </>
         )
       }
       ```
